Facet Wrap¶

facet_wrap allows you to pass a list of columns OR a catogorical column and create one plot for each column / each category

In [4]:
facet_wrap(
    sns.barplot,
    data=diamonds,
    facet=['cut','color'], # will wrap around the first argument by default, in this case x
    y = 'price', # is passed to the function as part of the kwargs
    height = 4.5, # subplot height
    width = 10, # subplot width
) 
plt.show()

In [6]:
facet_wrap(
    lambda y,**kwargs: sns.barplot(y=y,**kwargs), # if you want to facet on another argument then the first you can pass a lambda
    data=diamonds,
    facet=['depth','table','price','x','y','z'],
    x='cut',
    hue='color',
    height = 4.5, # subplot height
    width = 10, # subplot width
    col_wrap = 2, # when to start the next row
) 
plt.show()

you can also use facet_type='group' to create a single plot for each level of the facet column

In [7]:
facet_wrap(
    sns.barplot, 
    data=diamonds,
    facet = 'cut', 
    x = 'color',
    y='price' ,
    facet_type = 'group', # create a single plot for each level of the facet column
    height = 4.5, # subplot height
    width = 10, # subplot width
    col_wrap = 2, # when to start the next row,
) 
plt.subplots_adjust(hspace=.3)
plt.show()
